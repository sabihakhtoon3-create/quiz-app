  <script>
    let score = 0, total = 4;

    // Show first question
    document.getElementById("q1").style.display = "block";

    function check(button, isCorrect, currentQ, nextQ) {
      let parent = button.parentElement;
      let buttons = parent.querySelectorAll("button");
      let correctBtn = null;

      // Find the correct button in this question
      buttons.forEach(b => {
        if (b.getAttribute("onclick").includes("true")) {
          correctBtn = b;
        }
      });

      if (isCorrect) {
        button.classList.add("correct");
        score++;
      } else {
        button.classList.add("wrong");
        if (correctBtn) correctBtn.classList.add("highlight"); // Highlight correct answer
      }

      // Disable all buttons
      buttons.forEach(b => b.disabled = true);

      // Move to next question after 1.5 seconds
      setTimeout(() => {
        document.getElementById(currentQ).style.display = "none";
        document.getElementById(nextQ).style.display = "block";

        // If end, show score
        if (nextQ === "end") {
          document.getElementById("score").innerText =
            `You got ${score} out of ${total} correct.`;
        }
      }, 1500);
    }
  </script>
